# Python scripts
# Use the right arm to grasp bottle, move to the target location (x=0.2, y=0.1), and then open the gripper to release the object.

```python
from functools import partial
from embodichain.lab.sim.agent.atom_actions import *

# Step 1 — Reach and grasp the bottle
drive(
    left_arm_action=None,
    right_arm_action=grasp(
        robot_name="right_arm",
        obj_name="bottle",
    ),
    error_functions=[
        partial(
            inject_object_error,
            error_type="misplaced_object",
            error_obj="bottle",
        )
    ],
    monitor_sequences=[
        [
            partial(
                monitor_object_moved,
                obj_name="bottle",
                threshold=0.02,
            )
        ]
    ],
    recovery_sequences=[
        [
            partial(
                drive,
                left_arm_action=None,
                right_arm_action=partial(
                    grasp,
                    robot_name="right_arm",
                    obj_name="bottle",
                ),
            ),
        ]
    ],
)

# Step 2 — Move to target location
drive(
    left_arm_action=None,
    right_arm_action=move_to_absolute_position(
        robot_name="right_arm",
        x=0.2,
        y=0.1,
    ),
    error_functions=[
        partial(
            inject_object_error,
            error_type="misplaced_object",
            error_obj="bottle",
        )
    ],
    monitor_sequences=[
        [
            partial(
                monitor_object_held,
                robot_name="right_arm",
                obj_name="bottle",
                threshold=0.05,
            )
        ]
    ],
    recovery_sequences=[
        [
            partial(
                drive,
                left_arm_action=None,
                right_arm_action=partial(
                    grasp,
                    robot_name="right_arm",
                    obj_name="bottle",
                ),
            ),
            partial(
                drive,
                left_arm_action=None,
                right_arm_action=partial(
                    move_to_absolute_position,
                    robot_name="right_arm",
                    x=0.2,
                    y=0.1,
                ),
            ),
        ]
    ],
)

# Step 3 — Adjust to a nearby release pose while still holding the bottle
drive(
    left_arm_action=None,
    right_arm_action=move_to_absolute_position(
        robot_name="right_arm",
        x=0.25,
        y=0.05,
    ),
    error_functions=[
        partial(
            inject_action_error,
            error_type="wrong_affordance",
            error_arm="right_arm",
            relative_error_xyz=[0.02, 0.01, 0.0],
        )
    ],
    monitor_sequences=[
        [
            partial(
                monitor_object_held,
                robot_name="right_arm",
                obj_name="bottle",
                threshold=0.05,
            )
        ]
    ],
    recovery_sequences=[
        [
            partial(
                drive,
                left_arm_action=None,
                right_arm_action=partial(
                    grasp,
                    robot_name="right_arm",
                    obj_name="bottle",
                ),
            ),
            partial(
                drive,
                left_arm_action=None,
                right_arm_action=partial(
                    move_to_absolute_position,
                    robot_name="right_arm",
                    x=0.25,
                    y=0.05,
                ),
            ),
        ]
    ],
)

# Step 4 — Open gripper to release the object
drive(
    left_arm_action=None,
    right_arm_action=open_gripper(
        robot_name="right_arm",
    ),
)

# Step 5 — Return the arm to the initial pose
drive(
    left_arm_action=None,
    right_arm_action=back_to_initial_pose(
        robot_name="right_arm",
    ),
)
```

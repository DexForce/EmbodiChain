Atomic Actions#

EmbodiChain’s atomic action layer provides a high-level, composable interface for common manipulation primitives such as move, pick up, and place. Each action encapsulates the full planning pipeline — grasp-pose estimation, IK, trajectory generation, and gripper interpolation — behind a single execute() call, making it straightforward to chain multiple actions together into complex robot behaviours.

Key Features#

  • Semantic-aware execution — actions accept either a raw pose tensor or an ObjectSemantics descriptor that bundles affordance data (grasp poses, interaction points) with the simulation entity.

  • Three built-in primitives — MoveAction, PickUpAction, and PlaceAction cover the most common tabletop manipulation workflows out of the box. See the supported_atomic_actions table for configs and target types.

  • Extensible registry — custom actions can be registered globally with register_action and discovered by the engine at runtime.

  • Engine orchestration — AtomicActionEngine sequences multiple actions, threads start_qpos from one action to the next, and returns a single concatenated trajectory ready to replay in the simulator.

For the full design overview, architecture diagram, and extension guide see Atomic Actions.

 72def parse_arguments():
 73    """
 74    Parse command-line arguments to configure the simulation.
 75
 76    Returns:
 77        argparse.Namespace: Parsed arguments including number of environments, device, and rendering options.
 78    """
 79    parser = argparse.ArgumentParser(
 80        description="Create and simulate a robot in SimulationManager"
 81    )
 82    parser.add_argument(
 83        "--enable_rt", action="store_true", help="Enable ray tracing rendering"
 84    )
 85    parser.add_argument(
 86        "--num_envs", type=int, default=1, help="Number of parallel environments"
 87    )
 88    return parser.parse_args()

 91def initialize_simulation(args):
 92    """
 93    Initialize the simulation environment based on the provided arguments.
 94
 95    Args:
 96        args (argparse.Namespace): Parsed command-line arguments.
 97
 98    Returns:
 99        SimulationManager: Configured simulation manager instance.
100    """
101    config = SimulationManagerCfg(
102        headless=True,
103        sim_device="cuda",
104        enable_rt=args.enable_rt,
105        physics_dt=1.0 / 100.0,
106        num_envs=args.num_envs,
107    )
108    sim = SimulationManager(config)
109
110    light = sim.add_light(
111        cfg=LightCfg(uid="main_light", intensity=50.0, init_pos=(0, 0, 2.0))
112    )
113
114    return sim

117def create_robot(sim: SimulationManager, position=[0.0, 0.0, 0.0]):
118    """
119    Create and configure a robot with an arm and a dexterous hand in the simulation.
120
121    Args:
122        sim (SimulationManager): The simulation manager instance.
123
124    Returns:
125        Robot: The configured robot instance added to the simulation.
126    """
127    # Retrieve URDF paths for the robot arm and hand
128    ur10_urdf_path = get_data_path("UniversalRobots/UR10/UR10.urdf")
129    gripper_urdf_path = get_data_path("DH_PGC_140_50_M/DH_PGC_140_50_M.urdf")
130    # Configure the robot with its components and control properties
131    cfg = RobotCfg(
132        uid="UR10",
133        urdf_cfg=URDFCfg(
134            components=[
135                {"component_type": "arm", "urdf_path": ur10_urdf_path},
136                {"component_type": "hand", "urdf_path": gripper_urdf_path},
137            ]
138        ),
139        drive_pros=JointDrivePropertiesCfg(
140            stiffness={"JOINT[0-9]": 1e4, "FINGER[1-2]": 1e2},
141            damping={"JOINT[0-9]": 1e3, "FINGER[1-2]": 1e1},
142            max_effort={"JOINT[0-9]": 1e5, "FINGER[1-2]": 1e3},
143            drive_type="force",
144        ),
145        control_parts={
146            "arm": ["JOINT[0-9]"],
147            "hand": ["FINGER[1-2]"],
148        },
149        solver_cfg={
150            "arm": PytorchSolverCfg(
151                end_link_name="ee_link",
152                root_link_name="base_link",
153                tcp=[
154                    [0.0, 1.0, 0.0, 0.0],
155                    [-1.0, 0.0, 0.0, 0.0],
156                    [0.0, 0.0, 1.0, 0.12],
157                    [0.0, 0.0, 0.0, 1.0],
158                ],
159            )
160        },
161        init_qpos=[0.0, -np.pi / 2, -np.pi / 2, np.pi / 2, -np.pi / 2, 0.0, 0.0, 0.0],
162        init_pos=position,
163    )
164    return sim.add_robot(cfg=cfg)

167def create_mug(sim: SimulationManager) -> RigidObject:
168    mug_cfg = RigidObjectCfg(
169        uid="mug",
170        shape=MeshCfg(
171            fpath=get_data_path("CoffeeCup/cup.ply"),
172        ),
173        attrs=RigidBodyAttributesCfg(
174            mass=0.01,
175            dynamic_friction=0.97,
176            static_friction=0.99,
177        ),
178        max_convex_hull_num=16,
179        init_pos=[0.55, 0.0, 0.01],
180        init_rot=[0.0, 0.0, -90],
181        body_scale=(4, 4, 4),
182    )
183    mug = sim.add_rigid_object(cfg=mug_cfg)
184    return mug

187def main():
188    """Pick up a mug and place it at a new location using atomic actions."""
189    args = parse_arguments()
190
191    # ------------------------------------------------------------------ #
192    # Step 1: Set up simulation, robot, and object                        #
193    # ------------------------------------------------------------------ #
194    sim: SimulationManager = initialize_simulation(args)
195    robot = create_robot(sim)
196    mug = create_mug(sim)
197
198    # ------------------------------------------------------------------ #
199    # Step 2: Create a MotionGenerator for the robot                      #
200    # MotionGenerator handles trajectory planning (IK + TOPPRA smoothing) #
201    # ------------------------------------------------------------------ #
202    motion_gen = MotionGenerator(
203        cfg=MotionGenCfg(planner_cfg=ToppraPlannerCfg(robot_uid=robot.uid))
204    )
205
206    # ------------------------------------------------------------------ #
207    # Step 3: Configure the three atomic actions                          #
208    #                                                                     #
209    #  PickUpAction  — approach → close gripper → lift                   #
210    #  PlaceAction   — lower → open gripper → retract                    #
211    #  MoveAction    — free-space move to a target EEF pose               #
212    # ------------------------------------------------------------------ #
213    # Gripper joint values for this robot (DH_PGC_140):
214    #   open  = [0.00, 0.00]   (fully open)
215    #   close = [0.025, 0.025] (grasping width)
216    hand_open = torch.tensor([0.00, 0.00], dtype=torch.float32, device=sim.device)
217    hand_close = torch.tensor([0.025, 0.025], dtype=torch.float32, device=sim.device)
218
219    pickup_cfg = PickUpActionCfg(
220        control_part="arm",
221        hand_control_part="hand",
222        hand_open_qpos=hand_open,
223        hand_close_qpos=hand_close,
224        # Approach the object from directly above (negative world-Z)
225        approach_direction=torch.tensor(
226            [0.0, 0.0, -1.0], dtype=torch.float32, device=sim.device
227        ),
228        pre_grasp_distance=0.15,  # hover 15 cm above before descending
229        lift_height=0.15,  # lift 15 cm after grasping
230    )
231
232    place_cfg = PlaceActionCfg(
233        control_part="arm",
234        hand_control_part="hand",
235        hand_open_qpos=hand_open,
236        hand_close_qpos=hand_close,
237        lift_height=0.15,
238    )
239
240    move_cfg = MoveActionCfg(
241        control_part="arm",
242    )
243
244    # ------------------------------------------------------------------ #
245    # Step 4: Build the AtomicActionEngine                                #
246    #                                                                     #
247    # actions_cfg_list defines the ORDER of actions that execute_static() #
248    # will run. Each entry is matched positionally to target_list.        #
249    # ------------------------------------------------------------------ #
250    atomic_engine = AtomicActionEngine(
251        motion_generator=motion_gen,
252        actions_cfg_list=[pickup_cfg, place_cfg, move_cfg],
253    )
254
255    sim.init_gpu_physics()
256    sim.open_window()
257
258    # ------------------------------------------------------------------ #
259    # Step 5: Describe the mug with ObjectSemantics                       #
260    #                                                                     #
261    # ObjectSemantics bundles together:                                   #
262    #   - geometry (mesh vertices/triangles for grasp annotation)         #
263    #   - affordance (how to grasp the object — here antipodal grasps)   #
264    #   - entity reference (so the action can read the live object pose)  #
265    # ------------------------------------------------------------------ #
266    mug_grasp_affordance = AntipodalAffordance(
267        object_label="mug",
268        force_reannotate=False,
269        custom_config={
270            "gripper_collision_cfg": GripperCollisionCfg(
271                max_open_length=0.088, finger_length=0.078, point_sample_dense=0.012
272            ),
273            "generator_cfg": GraspGeneratorCfg(
274                viser_port=11801,
275                antipodal_sampler_cfg=AntipodalSamplerCfg(
276                    n_sample=20000, max_length=0.088, min_length=0.003
277                ),
278            ),
279        },
280    )
281    mug_semantics = ObjectSemantics(
282        label="mug",
283        geometry={
284            "mesh_vertices": mug.get_vertices(env_ids=[0], scale=True)[0],
285            "mesh_triangles": mug.get_triangles(env_ids=[0])[0],
286        },
287        affordance=mug_grasp_affordance,
288        entity=mug,  # needed so PickUpAction can read the mug's live pose
289    )
290
291    # ------------------------------------------------------------------ #
292    # Step 6: Define target poses for place and final rest                #
293    #                                                                     #
294    # Poses are 4×4 homogeneous transforms (rotation | translation).     #
295    # For PickUpAction the target is mug_semantics — the action computes  #
296    # the grasp pose automatically from the affordance.                   #
297    # ------------------------------------------------------------------ #
298    # Place the mug 20 cm to the left and 40 cm forward from its pickup pose
299    place_xpos = torch.tensor(
300        [
301            [-0.0539, -0.9985, -0.0022, 0.2489],
302            [-0.9977, 0.0540, -0.0401, 0.3970],
303            [0.0401, 0.0000, -0.9992, 0.2400],
304            [0.0000, 0.0000, 0.0000, 1.0000],
305        ],
306        dtype=torch.float32,
307        device=sim.device,
308    )
309
310    # Move the arm to a safe resting pose after placing
311    rest_xpos = torch.tensor(
312        [
313            [-0.0539, -0.9985, -0.0022, 0.5000],
314            [-0.9977, 0.0540, -0.0401, 0.0000],
315            [0.0401, 0.0000, -0.9992, 0.5000],
316            [0.0000, 0.0000, 0.0000, 1.0000],
317        ],
318        dtype=torch.float32,
319        device=sim.device,
320    )
321
322    # ------------------------------------------------------------------ #
323    # Step 7: Plan and execute the full sequence                          #
324    #                                                                     #
325    # execute_static() plans all three actions in order and returns a     #
326    # single concatenated joint trajectory (n_envs, n_waypoints, dof).   #
327    # We then replay it frame-by-frame in the simulator.                 #
328    # ------------------------------------------------------------------ #
329    print("Planning pick → place → move trajectory...")
330    is_success, traj = atomic_engine.execute_static(
331        target_list=[mug_semantics, place_xpos, rest_xpos]
332    )
333
334    if not is_success:
335        print("Planning failed. Check that the target poses are reachable.")
336        return
337
338    print(f"Success! Replaying {traj.shape[1]} waypoints...")
339    for i in range(traj.shape[1]):
340        robot.set_qpos(traj[:, i])
341        sim.update(step=4)
342        time.sleep(1e-2)
343
344    input("Press Enter to exit...")
345
346
347if __name__ == "__main__":
348    main()

Typical Usage#

Setting up the engine#

import torch
from embodichain.lab.sim.planners import MotionGenerator, MotionGenCfg
from embodichain.lab.sim.atomic_actions import (
    AtomicActionEngine,
    PickUpActionCfg,
    PlaceActionCfg,
    MoveActionCfg,
)

motion_gen = MotionGenerator(cfg=MotionGenCfg(...))

hand_open  = torch.tensor([0.00,  0.00],  dtype=torch.float32, device=device)
hand_close = torch.tensor([0.025, 0.025], dtype=torch.float32, device=device)

pickup_cfg = PickUpActionCfg(
    hand_open_qpos=hand_open,
    hand_close_qpos=hand_close,
    control_part="arm",
    hand_control_part="hand",
    approach_direction=torch.tensor([0.0, 0.0, -1.0], dtype=torch.float32, device=device),
    pre_grasp_distance=0.15,
    lift_height=0.15,
)
place_cfg = PlaceActionCfg(
    hand_open_qpos=hand_open,
    hand_close_qpos=hand_close,
    control_part="arm",
    hand_control_part="hand",
    lift_height=0.15,
)
move_cfg = MoveActionCfg(control_part="arm")

engine = AtomicActionEngine(
    motion_generator=motion_gen,
    actions_cfg_list=[pickup_cfg, place_cfg, move_cfg],
)

from embodichain.lab.sim.atomic_actions import (
    ObjectSemantics,
    AntipodalAffordance,

    label="mug",
    geometry={
        "mesh_vertices": mug.get_vertices(env_ids=[0], scale=True)[0],
        "mesh_triangles": mug.get_triangles(env_ids=[0])[0],
    },
    affordance=affordance,
    entity=mug,   # required so the action can query the live object pose
)

Registering custom actions#

from embodichain.lab.sim.atomic_actions import AtomicAction, ActionCfg, register_action

class PushAction(AtomicAction):
    def execute(self, target, start_qpos=None, **kwargs):
        # ... your planning logic ...
        return is_success, trajectory, joint_ids

    def validate(self, target, start_qpos=None, **kwargs):
        return True   # quick feasibility check

register_action("push", PushAction)

Notes & Best Practices#

  • PickUpAction expects an AntipodalAffordance with valid mesh data (mesh_vertices / mesh_triangles) so the grasp generator can annotate the object. Set force_reannotate=False (the default) to reuse cached annotations across episodes.

  • ObjectSemantics.entity must be set when using semantic targets so the action can read the object’s current world pose at planning time.

  • For static (non-physics) playback, iterate over trajectory[:, i] and call robot.set_qpos directly; for physics-enabled playback, feed waypoints through your controller or gym wrapper instead.

  • To add a new action type, see Atomic Actions.
